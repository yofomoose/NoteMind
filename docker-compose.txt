version: '3.8'

services:
  # Сервис MongoDB
  mongodb:
    image: mongo:latest
    container_name: notemind-mongodb
    ports:
      - "27017:27017"
    volumes:
      - ./data/db:/data/db
    environment:
      - MONGO_INITDB_DATABASE=notemind
    restart: unless-stopped

  # Опциональный сервис Redis (для кэширования и веб-сокетов)
  redis:
    image: redis:alpine
    container_name: notemind-redis
    ports:
      - "6379:6379"
    volumes:
      - ./data/redis:/data
    restart: unless-stopped

  # Сервис для бэкенда
  server:
    build:
      context: ./server
      dockerfile: Dockerfile
    container_name: notemind-server
    ports:
      - "5000:5000"
    depends_on:
      - mongodb
      - redis
    volumes:
      - ./server:/app
      - /app/node_modules
    environment:
      - NODE_ENV=development
      - PORT=5000
      - MONGO_URI=mongodb://mongodb:27017/notemind
      - REDIS_HOST=redis
      - REDIS_PORT=6379
    restart: unless-stopped

  # Сервис для фронтенда
  client:
    build:
      context: ./client
      dockerfile: Dockerfile
    container_name: notemind-client
    ports:
      - "3000:3000"
    depends_on:
      - server
    volumes:
      - ./client:/app
      - /app/node_modules
    environment:
      - REACT_APP_API_URL=http://localhost:5000
    restart: unless-stopped
